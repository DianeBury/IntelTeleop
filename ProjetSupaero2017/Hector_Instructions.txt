sudo apt-get install ros-kinetic-laser-geometry
sudo apt-get install ros-kinetic-tf-conversions
sudo apt-get install ros-kinetic-tf2-geometry-msgs
sudo apt-get install ros-kinetic-joy


/*** Sources hector à placer dans catkin_ws/src et à compiler ***/

git clone -b kinetic-devel https://github.com/tu-darmstadt-ros-pkg/hector_quadrotor
git clone -b catkin https://github.com/tu-darmstadt-ros-pkg/hector_localization
git clone -b kinetic-devel https://github.com/tu-darmstadt-ros-pkg/hector_gazebo
git clone -b kinetic-devel https://github.com/tu-darmstadt-ros-pkg/hector_models
git clone -b catkin https://github.com/tu-darmstadt-ros-pkg/hector_slam


/*** Le module hector_slam n'est en fait pas utile dans notre cadre,
 *** donc on va le dégager du projet
 *** ce qui n'empêche pas de garder les sources dans sun coin pour une utilisation future
 ***/

Supprimer (ou mettre ailleurs) le dossier
	(...)/IntelTeleop/ProjetSupaero2017/catkin_ws/src/hector_quadrotor/hector_slam

Dans le fichier
	/home/pie/hector_quadrotor/catkin_ws/src/hector_quadrotor/hector_quadrotor_gazebo/launch/spawn_quadrotor_slam.launch:
commenter ou supprimer la ligne
	<include file="$(find hector_slam_launch)/launch/tutorial.launch"/>


/*** On ne peut pas compiler avec MatLab
 *** On désactive les modules qui l'utilisent
 ***/

Dans le fichier
	/home/pie/IntelTeleop/ProjetSupaero2017/catkin_ws/src/hector_quadrotor/hector_quadrotor/hector_quadrotor_gazebo/urdf/CMakeLists.txt
Remplacer ON par OFF :
	option(USE_PROPULSION_PLUGIN "Use a model of the quadrotor propulsion system" ON)
	option(USE_AERODYNAMICS_PLUGIN "Use a model of the quadrotor aerodynamics" ON)

/*** Dégager les warnings (relatifs à hector) ***/

Dans le fichier
	hector_quadrotor/hector_quadrotor/hector_quadrotor_actions/CMakeLists.txt
Supprimer ou commenter les lignes
	add_dependencies(landing_action hector_uav_msgs_generate_message_cpp)
	add_dependencies(pose_action hector_uav_msgs_generate_message_cpp)
	add_dependencies(takeoff_action hector_uav_msgs_generate_message_cpp)

Dans le fichier
	hector_quadrotor/hector_quadrotor/hector_quadrotor_controllers/CMakeLists.txt
Supprimer ou commenter la ligne
	add_dependencies(hector_quadrotor_controllers hector_uav_msgs_generate_message_cpp)
